\section{2. Cơ sở lý thuyết về mạng WLAN}

\subsection{2.1 Khái niệm}
WLAN - Wireless Local Access Network là một hệ thống mạng máy tính sử dụng sóng vô tuyến thay vì sử dụng kết nối vật lí như cáp Ethernet. 
Mạng WLAN thuờng sử dụng trong một phạm vi nhỏ như tòa nhà, lớp học, căn tin,...\cite{kurose2017}

Thông qua mạng WLAN, các thiết bị đầu cuối như laptop, smartphone, desktop, ... có thể truyền hoặc nhận các gói dữ liệu liên tục và linh hoạt. 
Đây là điểm mạnh lớn vì nó cho phép người dùng có thể duy trì kết nối trong di chuyển hoặc thay đổi vị trí trong khu vực phủ sóng nhất định\cite{tanenbaum2010}.

Hiện nay hầu hết các mạng WLAN hoạt động dựa trên tiêu chuẩn IEEE 802.11\cite{10979691}, ta sẽ có hai mô hình mạng cơ bản:
\begin{itemize}
    \item \textbf{Mô hình mạng Infrastructure:} là mô hình mà các thiết bị đầu cuối kết nối và trao đổi dữ liệu với nhau thông qua một thiết bị trung tâm như AP.
    \item \textbf{Mô hình mạng Ad-hoc:} tương tự nhưng là theo cơ chế ngang hàng mà không cần thiết bị trung tâm.
\end{itemize}

\subsection{2.2 Các thành phần của mạng WLAN}
Mạng WLAN được cấu thành từ nhiều thiết bị và thành phần khác nhau, đóng vai trò quan trọng trong việc truyền tải và xử lý tín hiệu không dây:
\begin{itemize}
    \item \textbf{Access Point (AP) / Wireless Router:} là thiết bị trung tâm của mạng WLAN, 
    có chức năng thu phát tín hiệu vô tuyến và đóng vai trò cầu nối giữa các thiết bị không dây 
    với mạng có dây (LAN) hoặc Internet. Access Point hoạt động ở tầng liên kết dữ liệu (Data 
    Link Layer), quản lý việc kết nối, xác thực và phân phối địa chỉ IP cho các thiết bị đầu cuối 
    thông qua giao thức DHCP. Wireless Router tích hợp thêm chức năng định tuyến (routing), cho 
    phép kết nối trực tiếp với modem để chia sẻ đường truyền Internet đến toàn bộ mạng nội bộ.
    \item \textbf{Wireless Network Interface Controller (WNIC):} hay còn gọi là card mạng không 
    dây, là thành phần phần cứng được tích hợp hoặc gắn thêm vào thiết bị đầu cuối, đảm nhận 
    nhiệm vụ mã hóa dữ liệu thành tín hiệu vô tuyến để phát đi và giải mã tín hiệu nhận về thành 
    dữ liệu số. WNIC hoạt động theo các chuẩn IEEE 802.11 (Wi-Fi), hỗ trợ một hoặc nhiều băng tần 
    (2.4\,GHz, 5\,GHz, 6\,GHz), và thực hiện các cơ chế kiểm soát truy cập môi trường (MAC) như 
    CSMA/CA để tránh xung đột tín hiệu trong quá trình truyền.
    \item \textbf{Thiết bị đầu cuối (End Devices):} là các thiết bị của người dùng kết nối vào 
    mạng WLAN thông qua WNIC, bao gồm máy tính xách tay, điện thoại thông minh, máy tính bảng, 
    máy in không dây, camera IP và các thiết bị IoT. Đây là nguồn phát sinh và tiếp nhận dữ liệu 
    trong mạng; mỗi thiết bị đầu cuối được nhận diện bởi địa chỉ MAC duy nhất và địa chỉ IP do 
    AP/Router cấp phát, cho phép định tuyến gói tin chính xác trong hệ thống mạng.
    \item \textbf{Môi trường truyền dẫn:} trong mạng WLAN là sóng điện từ lan truyền trong không 
    gian tự do, chủ yếu sử dụng hai băng tần phổ biến:
        \begin{itemize}
            \item \textbf{Băng tần 2.4\,GHz:} có khả năng truyền tín hiệu xa hơn và xuyên qua vật 
            cản tốt hơn, tuy nhiên dễ bị nhiễu do phải chia sẻ băng tần với nhiều thiết bị khác như 
            lò vi sóng và Bluetooth. Băng tần này cung cấp 13 kênh (tại Việt Nam), trong đó chỉ có 
            3 kênh không chồng lấn (1, 6, 11), dẫn đến tốc độ truyền dữ liệu thấp hơn trong môi 
            trường đông thiết bị.
            \item \textbf{Băng tần 5\,GHz:} cung cấp băng thông rộng hơn với nhiều kênh không chồng 
            lấn hơn, cho tốc độ truyền dữ liệu cao hơn và ít nhiễu hơn. Tuy nhiên, tầm phủ sóng 
            ngắn hơn và khả năng xuyên vật cản kém hơn so với băng tần 2.4\,GHz, phù hợp với các 
            ứng dụng đòi hỏi tốc độ cao trong phạm vi gần.
        \end{itemize}
\end{itemize}

\subsection{2.3 Mô hình triển khai và Giải pháp kết nối}

\subsubsection{2.3.1 Mô hình Infrastructure (Có Access Point/Router trung tâm)\cite{kurose2017}}

\begin{itemize}
    \item \textbf{Khái niệm:}
    \begin{itemize}
        \item Mọi gói tin đều phải đi qua một thiết bị trung gian là Access Point (AP) 
        hoặc Router, các thiết bị đầu cuối không giao tiếp trực tiếp với nhau~\cite{ref1}.
        \item Thường được lắp đặt cố định để phục vụ lâu dài: mạng Wi-Fi gia đình, 
        văn phòng, giảng đường đại học, \ldots
    \end{itemize}

    \item \textbf{Thành phần chính~\cite{ref1}:}
    \begin{itemize}
        \item \textbf{Access Point (AP) / Router Wi-Fi:} 
        Thiết bị trung tâm điều phối toàn bộ lưu lượng trong mạng.
        \item \textbf{Thiết bị đầu cuối (Stations -- STA):} 
        Laptop, smartphone, PC có trang bị card mạng không dây.
        \item \textbf{Hạ tầng cáp có dây (Wired Backbone):} 
        Cáp Ethernet nối các AP lại với nhau và kết nối ra Internet.
    \end{itemize}

    \item \textbf{Giao thức áp dụng:}
    \begin{itemize}
        \item \textbf{IEEE 802.11:} 
        Tiêu chuẩn quy định cách truyền nhận dữ liệu qua sóng vô tuyến ở lớp vật lý~\cite{ref3}.
        \item \textbf{WPA2-PSK (AES):} 
        Xác thực mật khẩu và mã hóa dữ liệu lớp 2~\cite{ref1}.
        \item \textbf{DHCP:} 
        Tự động cấp địa chỉ IP cho thiết bị khi kết nối~\cite{ref1}.
    \end{itemize}

    \item \textbf{Ưu điểm~\cite{ref1,ref4}:}
    \begin{itemize}
        \item Dễ mở rộng: thêm thiết bị mới chỉ cần nhập đúng mật khẩu.
        \item Bảo mật và quản lý tốt: có thể cấu hình tường lửa, 
        mật khẩu mã hóa và kiểm soát danh sách thiết bị truy cập trên Router.
    \end{itemize}

    \item \textbf{Nhược điểm~\cite{ref1,ref4}:}
    \begin{itemize}
        \item Chi phí đầu tư ban đầu cao: phải mua Router, AP, switch 
        và hệ thống cáp Ethernet.
        \item Rủi ro \textit{Single Point of Failure}: nếu Router trung tâm 
        hỏng hoặc mất điện thì toàn bộ mạng ngừng hoạt động.
    \end{itemize}
\end{itemize}


\subsubsection{2.3.2 Mô hình Ad-hoc (Mạng ngang hàng)\cite{kurose2017}}

\begin{itemize}
    \item \textbf{Khái niệm:}
    \begin{itemize}
        \item Các thiết bị kết nối thẳng với nhau theo kiểu ngang hàng 
        (Peer-to-Peer), hình thành mạng tạm thời mà không cần 
        chuẩn bị hạ tầng trước~\cite{ref1}.
        \item Một số ứng dụng thực tế: AirDrop, Quick Share, 
        mạng cảm biến không dây (WSN), mạng di động tùy biến (MANET).
    \end{itemize}

    \item \textbf{Thành phần chính~\cite{ref1}:}
    \begin{itemize}
        \item Chỉ gồm các thiết bị đầu cuối tự liên kết trực tiếp với nhau.
        \item Không có AP hay Router trung gian.
    \end{itemize}

    \item \textbf{Giao thức áp dụng:}
    \begin{itemize}
        \item \textbf{IEEE 802.11 (chế độ IBSS):} 
        Quy định kiến trúc mạng không dây độc lập, không có trạm gốc~\cite{ref3}.
        \item \textbf{Static IP / APIPA:} 
        Các máy phải gán địa chỉ IP thủ công hoặc dùng APIPA 
        tự cấp địa chỉ tạm thời trong dải 169.254.x.x 
        do không có DHCP~\cite{ref1}.
    \end{itemize}

    \item \textbf{Ưu điểm~\cite{ref1,ref4}:}
    \begin{itemize}
        \item Không tốn chi phí phần cứng, triển khai nhanh ở bất kỳ đâu.
        \item Không bị sập toàn bộ: một máy rời mạng thì 
        các máy còn lại vẫn duy trì kết nối bình thường.
    \end{itemize}

    \item \textbf{Nhược điểm~\cite{ref1,ref4}:}
    \begin{itemize}
        \item Hiệu năng giảm khi có nhiều thiết bị: băng thông bị chia sẻ,
        dễ xảy ra nhiễu sóng và xung đột gói tin; 
        mỗi máy còn phải tự xử lý định tuyến nên tốn tài nguyên hệ thống hơn.
        \item Bảo mật kém hơn mô hình Infrastructure do không có 
        thiết bị trung tâm kiểm soát truy cập.
    \end{itemize}
\end{itemize}
